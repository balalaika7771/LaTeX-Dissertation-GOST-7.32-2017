% ПРИМЕР КОНФИГУРАЦИИ ПУТЕЙ
% Скопируйте этот файл в config/paths.tex и настройте пути под свой проект
%
% По умолчанию все источники находятся в папке src/ относительно корня репозитория LaTeX
% Вы можете изменить пути, указав абсолютные или относительные пути

% Путь к папке с главами (относительно корня репозитория LaTeX)
% По умолчанию: src/chapters
% Пример для другой структуры: \newcommand{\chapterspath}{../my-thesis/chapters}
\newcommand{\chapterspath}{src/chapters}

% Путь к папке с изображениями (относительно корня репозитория LaTeX)
% По умолчанию: src/images
% Пример для другой структуры: \newcommand{\imagespath}{../my-thesis/images}
\newcommand{\imagespath}{src/images}

% Путь к папке с приложениями (относительно корня репозитория LaTeX)
% По умолчанию: src/appendix
\newcommand{\appendixpath}{src/appendix}

% Путь к папке с библиографией (относительно корня репозитория LaTeX)
% По умолчанию: bibliography
% Пример для другой структуры: \newcommand{\bibliographypath}{../my-thesis/bibliography}
\newcommand{\bibliographypath}{bibliography}

% Команды для удобного подключения файлов
% Эти команды автоматически используют пути, указанные выше
\newcommand{\inputchapter}[1]{\input{\chapterspath/#1}}
\newcommand{\inputappendix}[1]{\input{\appendixpath/#1}}

% Команда для включения изображений с автоматическим путем
% Использование: \includegraphicspath[width=0.8\textwidth]{example_plot.png}
% Или просто: \includegraphicspath{example_plot.png}
\newcommand{\includegraphicspath}[2][]{%
  \ifx\relax#1\relax
    \includegraphics{\imagespath/#2}%
  \else
    \includegraphics[#1]{\imagespath/#2}%
  \fi
}

%

%   \inputchapter{introduction}      % вместо \input{chapters/introduction}
%   \inputchapter{chapter1}           % вместо \input{chapters/chapter1}
%   \inputappendix{appendix1}        % вместо \input{appendix/appendix1}
%
% В главах для изображений:
%   \includegraphicspath[width=0.8\textwidth]{my_image.png}
%   % автоматически будет искать в src/images/my_image.png
%
% Для изменения путей просто измените значения выше
